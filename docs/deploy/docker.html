# 常用docker

  • nginx https://hub.docker.com/_/nginx (opens new window)
docker run --name some-nginx -p 8080:80 -v /some/content:/usr/share/nginx/html:ro -d nginx
#   [option] host:container
# /etc/nginx/nginx.conf
# /usr/share/nginx/html

# Dockerfile

# FROM baseImage
FROM nginx:alpine
# COPY source dest
COPY dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/nginx.conf

# RUN echo $PWD
# EXPOSE ...port
EXPOSE 80

# vuepress 使用nginx 需要配置 config.js ==>base:'/'

# 导出默认配置 
docker cp tmp-nginx-container:/etc/nginx/nginx.conf /host/path/nginx.conf

# 默认配置
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;


events {
    worker_connections  1024;
}


http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    gzip  on;

    include /etc/nginx/conf.d/*.conf;
}


docker build -t nginx:v1 .
# . docker build 的上下文路径 
docker run -d -p 80:80 nginx:v1
